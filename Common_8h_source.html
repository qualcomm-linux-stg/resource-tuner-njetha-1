75// Helper Macros to set Request and SysResource attributes.
76#define SET_REQUEST_PRIORITY(properties, priority)({ \
77 int32_t retVal; \
78 if(properties < 0 || priority < 0 || priority >= NUMBER_OF_RQUEST_PRIORITIES) { \
79 retVal = -1; \
80 } else { \
81 retVal = (int32_t) ((properties | priority)); \
82 } \
83 retVal; \
84}) \
85
86#define ADD_ALLOWED_MODE(properties, mode)({ \
87 int32_t retVal; \
88 if(properties < 0 || mode < MODE_RESUME || mode > MODE_DOZE) { \
89 retVal = -1; \
90 } else { \
91 retVal = (int32_t) (properties | (((properties >> 8) | mode) << 8)); \
92 } \
93 retVal; \
94}) \
